CREATE TABLE tb_notice
   ("NOTICENO" NUMBER, 
	"NOTICETITLE" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"NOTICEDATE" DATE DEFAULT SYSDATE, 
	"NOTICEWRITER" VARCHAR2(15 BYTE) NOT NULL ENABLE, 
	"NOTICECONTENT" VARCHAR2(2000 BYTE), 
	"ORIGINAL_FILEPATH" VARCHAR2(300 BYTE), 
	"RENAME_FILEPATH" VARCHAR2(300 BYTE), 
	 CONSTRAINT "PK_NOTICE" PRIMARY KEY ("NOTICENO")
)
